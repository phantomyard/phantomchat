# P2P Mesh Network — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a WebRTC mesh network between contacts, tunneled through Tor, with signaling via Nostr relays and 3-level message routing (direct → forward → relay fallback).

**Architecture:** `mesh-signaling.ts` handles SDP exchange via gift-wrap on Nostr relays. `mesh-manager.ts` manages WebRTC DataChannel connections per contact. `message-router.ts` decides the best delivery path. All WebRTC traffic uses TURNS (relay-only ICE policy) through Tor.

**Tech Stack:** TypeScript, WebRTC (native browser API), NIP-17 gift-wrap for signaling, TURNS for relay-only ICE

**Spec:** `docs/superpowers/specs/2026-04-11-tor-ui-distributed-relay-mesh-design.md` — Section 5

**Depends on:** Mini-Relay plan (relay-store.ts, mini-relay.ts, mini-relay.worker.ts)

```typescript
  urls: string;
  username: string;
  credential: string;

// Default TURN server — must use TURNS (TLS on 443) for Tor compatibility
// Standard TURN uses UDP which Tor cannot proxy
const DEFAULT_TURN: TurnServerConfig = {
  urls: 'turns:turn.nostra.chat:443',
  username: 'nostra',
  credential: 'anonymous'
};

export function getRtcConfig(turnOverride?: TurnServerConfig): RTCConfiguration {
  const turn = turnOverride || DEFAULT_TURN;

  return {
    iceServers: [{
      urls: turn.urls,
      username: turn.username,
      credential: turn.credential
    }],
    // CRITICAL: 'relay' = only TURN candidates, no host/srflx (which leak IP)
    iceTransportPolicy: 'relay',
    bundlePolicy: 'max-bundle',
    rtcpMuxPolicy: 'require'
  };
}

export const DATA_CHANNEL_OPTIONS: RTCDataChannelInit = {
  ordered: true,
  // No maxRetransmits — we want reliable delivery for NIP-01
};

  type: 'webrtc-signal';
  action: 'offer' | 'answer' | 'ice-candidate';
  sdp?: string;
  candidate?: RTCIceCandidateInit;
  sessionId: string;
```

```typescript
import {SIGNAL_KIND, SignalMessage} from '@lib/nostra/webrtc-config';

interface SignalEventInput {
  action: 'offer' | 'answer' | 'ice-candidate';
  sdp?: string;
  candidate?: RTCIceCandidateInit;
  sessionId: string;
}

/**
 * Create a Nostr event body for WebRTC signaling.
 * This is the rumor content inside a NIP-17 gift-wrap.
 * The caller wraps this in gift-wrap and publishes to relay.
 */
export function createSignalEvent(input: SignalEventInput): {kind: number; content: string} {
  const signal: SignalMessage = {
    type: 'webrtc-signal',
    action: input.action,
    sessionId: input.sessionId
  };

  if(input.sdp) signal.sdp = input.sdp;
  if(input.candidate) signal.candidate = input.candidate;

  return {
    kind: SIGNAL_KIND,
    content: JSON.stringify(signal)
  };
}

/**
 * Parse signaling content from a decrypted gift-wrap.
 * Returns null if the content is not a WebRTC signal.
 */
export function parseSignalContent(content: string): SignalMessage | null {
  try {
    const parsed = JSON.parse(content);
    if(parsed.type !== 'webrtc-signal') return null;
    if(!parsed.action || !parsed.sessionId) return null;
    return parsed as SignalMessage;
  } catch {
    return null;
  }
}

/**
 * Check if a decrypted rumor kind is a signaling message.
 */
export function isSignalKind(kind: number): boolean {
  return kind === SIGNAL_KIND;
}
```

```typescript
import {getRtcConfig, DATA_CHANNEL_NAME, DATA_CHANNEL_OPTIONS, SignalMessage} from '@lib/nostra/webrtc-config';
import {createSignalEvent, parseSignalContent} from '@lib/nostra/mesh-signaling';

const MAX_CONNECTIONS = 50;
const PING_INTERVAL = 30000; // 30s
const PING_TIMEOUT = 90000;  // 90s
const RECONNECT_DELAYS = [1000, 2000, 4000, 10000]; // burst then steady

type PeerStatus = 'connected' | 'connecting' | 'disconnected';

interface PeerState {
  pubkey: string;
  connection: RTCPeerConnection | null;
  dataChannel: RTCDataChannel | null;
  status: PeerStatus;
  lastSeen: number;
  latency: number;
  sessionId: string;
  reconnectAttempts: number;
  reconnectTimer: ReturnType<typeof setTimeout> | null;
  pingTimer: ReturnType<typeof setInterval> | null;
}

interface MeshCallbacks {
  sendSignal: (recipientPubkey: string, signal: {kind: number; content: string}) => Promise<void>;
  onPeerMessage: (pubkey: string, message: string) => void;
  onPeerConnected: (pubkey: string) => void;
  onPeerDisconnected: (pubkey: string) => void;
}

export class MeshManager {
  private peers: Map<string, PeerState> = new Map();
  private callbacks: MeshCallbacks;

  constructor(callbacks: MeshCallbacks) {
    this.callbacks = callbacks;
  }

  async connect(pubkey: string): Promise<void> {
    if(this.peers.size >= MAX_CONNECTIONS) {
      throw new Error('Max connections reached');
    }

    const existing = this.peers.get(pubkey);
    if(existing && existing.status !== 'disconnected') return;

    const sessionId = crypto.randomUUID();
    const config = getRtcConfig();
    const pc = new RTCPeerConnection(config);
    const dc = pc.createDataChannel(DATA_CHANNEL_NAME, DATA_CHANNEL_OPTIONS);

    const state: PeerState = {
      pubkey,
      connection: pc,
      dataChannel: dc,
      status: 'connecting',
      lastSeen: 0,
      latency: -1,
      sessionId,
      reconnectAttempts: 0,
      reconnectTimer: null,
      pingTimer: null
    };

    this.peers.set(pubkey, state);
    this.setupDataChannel(dc, state);
    this.setupConnection(pc, state);

    // Create and send offer
    const offer = await pc.createOffer();
    await pc.setLocalDescription(offer);

    const signal = createSignalEvent({
      action: 'offer',
      sdp: offer.sdp,
      sessionId
    });

    await this.callbacks.sendSignal(pubkey, signal);
  }

  async handleSignal(fromPubkey: string, content: string, rumorKind?: number): Promise<void> {
    const signal = parseSignalContent(content);
    if(!signal) return;

    switch(signal.action) {
      case 'offer':
        await this.handleOffer(fromPubkey, signal);
        break;
      case 'answer':
        await this.handleAnswer(fromPubkey, signal);
        break;
      case 'ice-candidate':
        await this.handleIceCandidate(fromPubkey, signal);
        break;
    }
  }

  private async handleOffer(fromPubkey: string, signal: SignalMessage) {
    const config = getRtcConfig();
    const pc = new RTCPeerConnection(config);

    const state: PeerState = {
      pubkey: fromPubkey,
      connection: pc,
      dataChannel: null,
      status: 'connecting',
      lastSeen: 0,
      latency: -1,
      sessionId: signal.sessionId,
      reconnectAttempts: 0,
      reconnectTimer: null,
      pingTimer: null
    };

    this.peers.set(fromPubkey, state);
    this.setupConnection(pc, state);

    // Listen for DataChannel from offerer
    pc.addEventListener('datachannel', (e: RTCDataChannelEvent) => {
      state.dataChannel = e.channel;
      this.setupDataChannel(e.channel, state);
    });

    await pc.setRemoteDescription(new RTCSessionDescription({type: 'offer', sdp: signal.sdp!}));
    const answer = await pc.createAnswer();
    await pc.setLocalDescription(answer);

    const answerSignal = createSignalEvent({
      action: 'answer',
      sdp: answer.sdp,
      sessionId: signal.sessionId
    });

    await this.callbacks.sendSignal(fromPubkey, answerSignal);
  }

  private async handleAnswer(fromPubkey: string, signal: SignalMessage) {
    const state = this.peers.get(fromPubkey);
    if(!state || !state.connection) return;
    if(state.sessionId !== signal.sessionId) return;

    await state.connection.setRemoteDescription(
      new RTCSessionDescription({type: 'answer', sdp: signal.sdp!})
    );
  }

  private async handleIceCandidate(fromPubkey: string, signal: SignalMessage) {
    const state = this.peers.get(fromPubkey);
    if(!state || !state.connection) return;

    if(signal.candidate) {
      await state.connection.addIceCandidate(signal.candidate);
    }
  }

  private setupConnection(pc: RTCPeerConnection, state: PeerState) {
    pc.addEventListener('icecandidate', async(e: RTCPeerConnectionIceEvent) => {
      if(e.candidate) {
        const signal = createSignalEvent({
          action: 'ice-candidate',
          candidate: e.candidate.toJSON(),
          sessionId: state.sessionId
        });
        await this.callbacks.sendSignal(state.pubkey, signal);
      }
    });

    pc.addEventListener('connectionstatechange', () => {
      if(pc.connectionState === 'failed' || pc.connectionState === 'disconnected') {
        this.handleDisconnect(state);
      }
    });
  }

  private setupDataChannel(dc: RTCDataChannel, state: PeerState) {
    dc.addEventListener('open', () => {
      state.status = 'connected';
      state.lastSeen = Date.now();
      state.reconnectAttempts = 0;
      this.startPing(state);
      this.callbacks.onPeerConnected(state.pubkey);
    });

    dc.addEventListener('close', () => {
      this.handleDisconnect(state);
    });

    dc.addEventListener('message', (e: MessageEvent) => {
      const data = e.data as string;

      // Handle ping/pong
      if(data === 'PING') {
        dc.send('PONG');
        return;
      }
      if(data === 'PONG') {
        state.lastSeen = Date.now();
        return;
      }

      state.lastSeen = Date.now();
      this.callbacks.onPeerMessage(state.pubkey, data);
    });
  }

  private startPing(state: PeerState) {
    if(state.pingTimer) clearInterval(state.pingTimer);

    state.pingTimer = setInterval(() => {
      if(!state.dataChannel || state.dataChannel.readyState !== 'open') return;

      // Check timeout
      if(Date.now() - state.lastSeen > PING_TIMEOUT) {
        this.handleDisconnect(state);
        return;
      }

      state.dataChannel.send('PING');
    }, PING_INTERVAL);
  }

  private handleDisconnect(state: PeerState) {
    const wasConnected = state.status === 'connected';
    state.status = 'disconnected';

    if(state.pingTimer) { clearInterval(state.pingTimer); state.pingTimer = null; }
    if(state.connection) { state.connection.close(); state.connection = null; }
    if(state.dataChannel) { state.dataChannel.close(); state.dataChannel = null; }

    if(wasConnected) {
      this.callbacks.onPeerDisconnected(state.pubkey);
    }

    // Reconnect with backoff
    const delay = RECONNECT_DELAYS[Math.min(state.reconnectAttempts, RECONNECT_DELAYS.length - 1)];
    state.reconnectAttempts++;

    state.reconnectTimer = setTimeout(() => {
      this.connect(state.pubkey).catch(() => {}); // retry silently
    }, delay);
  }

  disconnect(pubkey: string) {
    const state = this.peers.get(pubkey);
    if(!state) return;

    if(state.reconnectTimer) clearTimeout(state.reconnectTimer);
    if(state.pingTimer) clearInterval(state.pingTimer);
    if(state.connection) state.connection.close();
    if(state.dataChannel) state.dataChannel.close();

    state.reconnectAttempts = Infinity; // prevent auto-reconnect
    this.peers.delete(pubkey);
  }

  disconnectAll() {
    for(const [pubkey] of this.peers) {
      this.disconnect(pubkey);
    }
  }

  send(pubkey: string, message: string): boolean {
    const state = this.peers.get(pubkey);
    if(!state || !state.dataChannel || state.dataChannel.readyState !== 'open') {
      return false;
    }
    state.dataChannel.send(message);
    return true;
  }

  getStatus(pubkey: string): PeerStatus {
    return this.peers.get(pubkey)?.status ?? 'disconnected';
  }

  getConnectedPeers(): string[] {
    const result: string[] = [];
    for(const [pubkey, state] of this.peers) {
      if(state.status === 'connected') result.push(pubkey);
    }
    return result;
  }

  getPeerLatency(pubkey: string): number {
    return this.peers.get(pubkey)?.latency ?? -1;
  }
}
```

```typescript
import type {MeshManager} from '@lib/nostra/mesh-manager';
import type {NostrEvent} from '@lib/nostra/relay-store';

export type RoutePath = 'mesh-direct' | 'mesh-forward' | 'relay-external';

export interface RouteResult {
  path: RoutePath;
  delivered: boolean;
  forwardedVia?: string;
}

interface RouterDeps {
  meshManager: Pick<MeshManager, 'getStatus' | 'getConnectedPeers' | 'send'>;
  relayPublish: (event: NostrEvent) => Promise<boolean>;
  getContactsForPeer: (pubkey: string) => string[];
}

export class MessageRouter {
  private deps: RouterDeps;

  constructor(deps: RouterDeps) {
    this.deps = deps;
  }

  async route(event: NostrEvent, recipientPubkey: string): Promise<RouteResult> {
    let result: RouteResult;

    // Level 1: Direct mesh
    if(this.deps.meshManager.getStatus(recipientPubkey) === 'connected') {
      const sent = this.deps.meshManager.send(
        recipientPubkey,
        JSON.stringify(['EVENT', event])
      );
      if(sent) {
        result = {path: 'mesh-direct', delivered: true};
        // Also publish to relay as backup (belt and suspenders)
        this.deps.relayPublish(event).catch(() => {});
        return result;
      }
    }

    // Level 2: Forward via mutual contacts
    const mutualContacts = this.deps.getContactsForPeer(recipientPubkey);
    const connectedPeers = this.deps.meshManager.getConnectedPeers();

    for(const contactPubkey of mutualContacts) {
      if(connectedPeers.includes(contactPubkey)) {
        const sent = this.deps.meshManager.send(
          contactPubkey,
          JSON.stringify(['EVENT', event])
        );
        if(sent) {
          result = {path: 'mesh-forward', delivered: true, forwardedVia: contactPubkey};
          // Also publish to relay as backup
          this.deps.relayPublish(event).catch(() => {});
          return result;
        }
      }
    }

    // Level 3: Relay fallback
    const published = await this.deps.relayPublish(event);
    result = {path: 'relay-external', delivered: published};
    return result;
  }
}
```

```typescript
    return false;
```

```typescript
    const contactsSection = new SettingSection({name: 'Contacts'});
    this.contactListEl = document.createElement('div');
    this.contactListEl.classList.add('mesh-contact-list');
    contactsSection.content.append(this.contactListEl);

    this.scrollable.append(
      toggleSection.container,
      statsSection.container,
      contactsSection.container
    );

    // Update display
    this.updateDisplay();
    this.updateInterval = setInterval(() => this.updateDisplay(), 5000);

    rootScope.addEventListener('nostra_mesh_peer_connected', () => this.updateDisplay());
    rootScope.addEventListener('nostra_mesh_peer_disconnected', () => this.updateDisplay());

    this.onCleanup(() => {
      clearInterval(this.updateInterval);
    });
  }

  private updateDisplay() {
    const meshManager = (window as any).__nostraMeshManager;
    if(!meshManager) {
      this.statsEl.textContent = 'Mesh not initialized';
      return;
    }

    const connected = meshManager.getConnectedPeers();
    const contacts = (window as any).__nostraContacts || [];
    this.statsEl.textContent = `Connected to ${connected.length}/${contacts.length} contacts`;

    // Render contact list
    this.contactListEl.innerHTML = '';
    for(const pubkey of contacts) {
      const status = meshManager.getStatus(pubkey);
      const row = document.createElement('div');
      row.classList.add('mesh-contact-row');

      const name = document.createElement('span');
      name.classList.add('mesh-contact-name');
      name.textContent = pubkey.slice(0, 12) + '...';

      const statusText = document.createElement('span');
      statusText.classList.add('mesh-contact-status');
      statusText.textContent = status;

      row.append(dot, name, statusText);
      this.contactListEl.append(row);
    }
  }
```
