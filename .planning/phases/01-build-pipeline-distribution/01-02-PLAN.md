// from 'solid-transition-group'
export interface TransitionProps {
  children?: JSX.Element;
  mode?: string;            // 'out-in' | 'in-out' supported
  name?: string;            // CSS class prefix e.g. 'slide-fade' → .slide-fade-enter, .slide-fade-exit
  class?: string;
  enterClass?: string;      // override enter class (default: {name}-enter)
  exitClass?: string;       // override exit class (default: {name}-exit)
  enterActiveClass?: string; // active during enter (default: {name}-enter-active)
  exitActiveClass?: string;  // active during exit (default: {name}-exit-active)
  onBeforeEnter?: (el: Element) => void;
  onEnter?: (el: Element, done: () => void) => void;
  onAfterEnter?: (el: Element) => void;
  onBeforeExit?: (el: Element) => void;
  onExit?: (el: Element, done: () => void) => void;
  onAfterExit?: (el: Element) => void;
}

export const Transition: Component<TransitionProps>;
export const CSSTransition: Component<TransitionProps>;
export const TransitionGroup: Component<TransitionProps & {tag?: string}>;
