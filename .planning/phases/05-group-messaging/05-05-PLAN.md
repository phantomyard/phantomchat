Create `src/components/sidebarRight/tabs/nostraGroupInfo.ts` as a tweb `SliderSuperTab`.
This is a `.ts` file (NOT `.tsx`) using the plain DOM API with `Row`, `SettingSection`, etc.

The tab displays:
1. Group name as title (use `this.setTitle()` with the group name as a raw string cast to `LangPackKey`)
2. Section "Members" (`new SettingSection({name: 'Members' as LangPackKey, nameArgs: [memberCount]})`)
3. One `Row` per member:
   - title: display name from virtual-peers-db getAllMappings (or truncated pubkey fallback "P2P XXXXXX")
   - subtitle: "admin" for the admin member, empty for others
   - For non-admin members when current user is admin: attach a "Remove" context action
4. "Leave Group" button row at the bottom (`isDanger: true` button styled with SettingSection)
5. If current user is admin: also show a "Delete Group" button (calls leaveGroup for now — full delete is v2)

Leave Group flow:
- Tap "Leave Group" row
- Show `confirmationPopup` with: titleLangKey describing "Leave {GroupName}?", button text "Leave Group" (isDanger: true)
- On confirm: call `getGroupAPI().leaveGroup(groupId)`, then `this.close()`

Remove Member flow (admin only):
- Long-press or click on a member row calls `confirmationPopup` with "Remove {Name}?"
- On confirm: call `getGroupAPI().removeMember(groupId, memberPubkey)`, then re-render members

Use this structure:

```typescript
import {SliderSuperTab} from '@components/slider';
import SettingSection from '@components/settingSection';
import Row from '@components/row';
import {attachClickEvent} from '@helpers/dom/clickEvent';
import confirmationPopup from '@components/confirmationPopup';
import {getGroupAPI} from '@lib/nostra/group-api';
import {getGroupStore} from '@lib/nostra/group-store';
import {getAllMappings} from '@lib/nostra/virtual-peers-db';

export default class AppNostraGroupInfoTab extends SliderSuperTab {
  public groupPeerId: number;
  private groupId: string;

  public async init() {
    const store = getGroupStore();
    const group = await store.getByPeerId(this.groupPeerId);
    if(!group) {
      this.setTitle('Group' as any);
      return;
    }

    this.groupId = group.groupId;
    this.container.classList.add('nostra-group-info-container');
    this.setTitle(group.name as any);

    // Load member display names
    const allMappings = await getAllMappings();
    const mappingByPubkey = new Map(allMappings.map(m => [m.pubkey, m.displayName || 'P2P ' + m.pubkey.slice(0, 6).toUpperCase()]));

    // Determine own pubkey for admin check
    let ownPubkey: string | null = null;
    try {
      const {loadIdentity} = await import('@lib/nostra/nostr-key-storage');
      const identity = await loadIdentity();
      ownPubkey = identity?.publicKey ?? null;
    } catch{}

    const isAdmin = ownPubkey === group.adminPubkey;

    // Members section
    const membersSection = new SettingSection({
      name: 'Members' as any,
      nameArgs: [group.members.length]
    });

    for(const pubkey of group.members) {
      const displayName = mappingByPubkey.get(pubkey) || 'P2P ' + pubkey.slice(0, 6).toUpperCase();
      const isAdminMember = pubkey === group.adminPubkey;

      const row = new Row({
        title: displayName,
        subtitle: isAdminMember ? 'admin' : undefined,
        listenerSetter: this.listenerSetter
      });

      // Admin can remove non-admin members (but not themselves)
      if(isAdmin && !isAdminMember && pubkey !== ownPubkey) {
        attachClickEvent(row.container, async() => {
          await confirmationPopup({
            titleLangKey: 'AreYouSure' as any,
            descriptionLangKey: 'NostraRemoveMember' as any,
            button: {langKey: 'Remove' as any, isDanger: true}
          }).then(async() => {
            await getGroupAPI().removeMember(this.groupId, pubkey);
            row.container.remove();
          }).catch(() => {});
        }, {listenerSetter: this.listenerSetter});
      }

      membersSection.content.append(row.container);
    }

    this.scrollable.append(membersSection.container);

    // Leave group section
    const leaveSection = new SettingSection({noDelimiter: true});
    const leaveRow = new Row({
      title: 'Leave Group',
      titleElement: (() => {
        const el = document.createElement('span');
        el.style.color = 'var(--danger-color)';
        el.textContent = 'Leave Group';
        return el;
      })(),
      listenerSetter: this.listenerSetter
    });

    attachClickEvent(leaveRow.container, async() => {
      await confirmationPopup({
        titleLangKey: 'LeaveGroup' as any,
        button: {langKey: 'LeaveGroup' as any, isDanger: true}
      }).then(async() => {
        await getGroupAPI().leaveGroup(this.groupId);
        this.close();
      }).catch(() => {});
    }, {listenerSetter: this.listenerSetter});

    leaveSection.content.append(leaveRow.container);
    this.scrollable.append(leaveSection.container);
  }
}
```
